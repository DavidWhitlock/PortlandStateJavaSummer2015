001package edu.pdx.cs410J.whitlock;
002
003import edu.pdx.cs410J.InvokeMainTestCase;
004import org.junit.Ignore;
005import org.junit.Test;
006
007import java.util.ArrayList;
008
009import static org.hamcrest.CoreMatchers.startsWith;
010import static org.hamcrest.MatcherAssert.assertThat;
011import static org.hamcrest.core.IsEqual.equalTo;
012import static org.hamcrest.core.StringContains.containsString;
013
014/**
015 * JUnit tests for the Student class.  These tests extend <code>InvokeMainTestCase</code>
016 * which allows them to easily invoke the <code>main</code> method of <code>Student</code>.
017 * They also make use of the <a href="http://hamcrest.org/JavaHamcrest/">hamcrest</a>matchers
018 * for more readable assertion statements.
019 */
020public class StudentTest extends InvokeMainTestCase
021{
022
023  @Test
024  public void allStudentsSayThisClassIsTooMuchWork() {
025    Student student = new Student("name", new ArrayList(), 1.23, "male");
026    assertThat(student.says(), equalTo("This class is too much work"));
027  }
028
029  @Ignore
030  @Test
031  public void studentToStringOfDavemale3_64AlgorithmsOperatingSystemsJava() {
032    Student dave = createStudent("Dave");
033
034    assertThat(dave.toString(), equalTo("Dave has a GPA of 3.64 and is taking 3 classes: Algorithms, Operating Systems, and Java. He says \"This class is too much work\"."));
035  }
036
037  @Test
038  public void studentNamedDaveHasNameOfDave() {
039    Student dave = createStudent("Dave");
040
041    assertThat(dave.toString(), startsWith("Dave"));
042  }
043
044  @Test
045  public void studentNamedFrankHasNameOfFrank() {
046    Student frank = createStudent("Frank");
047
048    assertThat(frank.toString(), startsWith("Frank"));
049  }
050
051  @Test
052  public void studentWithGPAof314(){
053    double gpa = 3.14;
054    String expectedGpaStringValue = String.valueOf(gpa);
055
056    assertGpaIsAsExpected(gpa, expectedGpaStringValue);
057  }
058
059  private void assertGpaIsAsExpected(double gpa, String expectedGpaStringValue) {
060    Student aStudent = createStudent("student", gpa);
061    assertThat(aStudent.toString(), containsString(" has a GPA of " + expectedGpaStringValue));
062  }
063
064  @Test
065  public void wholeNumberGPAPrintsWithTwoDecimalPlace() {
066    double gpa = 4;
067    String expectedGpaStringValue = "4.0";
068
069    assertGpaIsAsExpected(gpa, expectedGpaStringValue);
070  }
071
072  @Test
073  public void gpaOf3point999DisplaysAs4point00() {
074    assertGpaIsAsExpected(3.999, "4.0");
075  }
076
077  /**
078   * Creates a new <code>Student</code> with the given name.
079   * @param name The name of the student
080   * @return The new student
081   */
082  private Student createStudent(String name) {
083    return createStudent(name, 3.64);
084  }
085
086  /**
087   * Creates a new <code>Student</code> with the given name.
088   * @param name The name of the student
089   * @param gpa
090   * @return The new student
091   */
092  private Student createStudent(String name, double gpa) {
093    return createStudent(name, gpa, 3);
094  }
095
096  /**
097   * Creates a new <code>Student</code> with the given name.
098   * @param name The name of the student
099   * @param gpa
100   * @param numClasses
101   * @return The new student
102   */
103  private Student createStudent(String name, double gpa, int numClasses) {
104    ArrayList classes = new ArrayList();
105
106    for (int i = 0; i < numClasses; i++) {
107      classes.add("Class " + i);
108    }
109
110    return new Student(name, classes, gpa, "male");
111  }
112
113  @Test
114  public void studentWhoTakesThreeClassesSpecifiesTheyTake3Classes(){
115    Student aStudent = createStudent("student", 3.14, 3);
116    assertThat(aStudent.toString(), containsString(" and is taking 3 classes:"));
117
118  }
119
120  @Test
121  public void invokingMainWithNoArgumentsHasExitCodeOf1() {
122    MainMethodResult result = invokeMain(Student.class);
123    assertThat(result.getExitCode(), equalTo(1));
124  }
125
126  @Test
127  public void invokingMainWithNoArgumentsPrintsMissingArgumentsToStandardError() {
128    MainMethodResult result = invokeMain(Student.class);
129    assertThat(result.getErr(), containsString("Missing command line arguments"));
130  }
131
132  @Test
133  public void invokingMainWithOnlyOneArgumentPrintsMissingGenderToStandardError() {
134    MainMethodResult result = invokeMain(Student.class, "Dave");
135    assertThat(result.getErr(), containsString("Missing gender"));
136    assertThat(result.getExitCode(), equalTo(1));
137  }
138
139  @Test
140  public void unparsableGPAResultsInErrorMessage() {
141    MainMethodResult result = invokeMain(Student.class, "Dave", "male", "BADGPA");
142    assertThat(result.getErr(), containsString("Invalid GPA: BADGPA"));
143    assertThat(result.getExitCode(), equalTo(1));
144  }
145
146}
