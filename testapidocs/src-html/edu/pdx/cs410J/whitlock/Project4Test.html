001package edu.pdx.cs410J.whitlock;
002
003import edu.pdx.cs410J.InvokeMainTestCase;
004import org.junit.FixMethodOrder;
005import org.junit.Test;
006import org.junit.runners.MethodSorters;
007
008import static org.hamcrest.MatcherAssert.assertThat;
009import static org.hamcrest.Matchers.containsString;
010import static org.hamcrest.Matchers.equalTo;
011
012/**
013 * Tests the {@link Project4} class by invoking its main method with various arguments 
014 */
015@FixMethodOrder(MethodSorters.NAME_ASCENDING)
016public class Project4Test extends InvokeMainTestCase {
017    private static final String HOSTNAME = "localhost";
018    private static final String PORT = System.getProperty("http.port", "8080");
019
020    @Test
021    public void test1NoCommandLineArguments() {
022        MainMethodResult result = invokeMain( Project4.class );
023        assertThat(result.getExitCode(), equalTo(1));
024        assertThat(result.getErr(), containsString(Project4.MISSING_ARGS));
025    }
026
027    @Test
028    public void test2EmptyServer() {
029        MainMethodResult result = invokeMain( Project4.class, HOSTNAME, PORT );
030        assertThat(result.getErr(), result.getExitCode(), equalTo(0));
031        String out = result.getOut();
032        assertThat(out, out, containsString(Messages.getMappingCount(0)));
033    }
034
035    @Test
036    public void test3NoValues() {
037        String key = "KEY";
038        MainMethodResult result = invokeMain( Project4.class, HOSTNAME, PORT, key );
039        assertThat(result.getErr(), result.getExitCode(), equalTo(0));
040        String out = result.getOut();
041        assertThat(out, out, containsString(Messages.getMappingCount(0)));
042        assertThat(out, out, containsString(Messages.formatKeyValuePair(key, null)));
043    }
044
045    @Test
046    public void test4AddValue() {
047        String key = "KEY";
048        String value = "VALUE";
049
050        MainMethodResult result = invokeMain( Project4.class, HOSTNAME, PORT, key, value );
051        assertThat(result.getErr(), result.getExitCode(), equalTo(0));
052        String out = result.getOut();
053        assertThat(out, out, containsString(Messages.mappedKeyValue(key, value)));
054
055        result = invokeMain( Project4.class, HOSTNAME, PORT, key );
056        out = result.getOut();
057        assertThat(out, out, containsString(Messages.getMappingCount(1)));
058        assertThat(out, out, containsString(Messages.formatKeyValuePair(key, value)));
059
060        result = invokeMain( Project4.class, HOSTNAME, PORT );
061        out = result.getOut();
062        assertThat(out, out, containsString(Messages.getMappingCount(1)));
063        assertThat(out, out, containsString(Messages.formatKeyValuePair(key, value)));
064    }
065}
