001package edu.pdx.cs410J.whitlock.client;
002
003import edu.pdx.cs410J.AbstractPhoneCall;
004import edu.pdx.cs410J.AbstractPhoneBill;
005
006import java.lang.Override;
007import java.util.ArrayList;
008import java.util.Collection;
009
010public class PhoneBill extends AbstractPhoneBill
011{
012  private String customerName;
013  private Collection<AbstractPhoneCall> calls = new ArrayList<AbstractPhoneCall>();
014
015  public PhoneBill(String customerName) {
016    this.customerName = customerName;
017
018  }
019
020  public PhoneBill() {
021  }
022
023  @Override
024  public String getCustomer() {
025    return this.customerName;
026  }
027
028  @Override
029  public void addPhoneCall(AbstractPhoneCall call) {
030    this.calls.add(call);
031  }
032
033  @Override
034  public Collection getPhoneCalls() {
035    return this.calls;
036  }
037}
