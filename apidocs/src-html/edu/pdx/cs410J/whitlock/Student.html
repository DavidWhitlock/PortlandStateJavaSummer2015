001package edu.pdx.cs410J.whitlock;
002
003import edu.pdx.cs410J.lang.Human;
004
005import java.text.DecimalFormat;
006import java.text.NumberFormat;
007import java.util.ArrayList;
008
009/**                                                                                 
010 * This class is represents a <code>Student</code>.                                 
011 */                                                                                 
012public class Student extends Human {
013
014  private double gpa;
015  private ArrayList<String> classes;
016
017  /**
018   * Creates a new <code>Student</code>                                             
019   *                                                                                
020   * @param name                                                                    
021   *        The student's name                                                      
022   * @param classes                                                                 
023   *        The names of the classes the student is taking.  A student              
024   *        may take zero or more classes.                                          
025   * @param gpa                                                                     
026   *        The student's grade point average                                       
027   * @param gender                                                                  
028   *        The student's gender ("male" or "female", case insensitive)             
029   */                                                                               
030  public Student(String name, ArrayList classes, double gpa, String gender) {
031    super(name);
032    this.gpa = gpa;
033    this.classes = classes;
034  }
035
036  /**                                                                               
037   * All students say "This class is too much work"                                 
038   */
039  @Override
040  public String says() {                                                            
041    return "This class is too much work";
042  }
043                                                                                    
044  /**                                                                               
045   * Returns a <code>String</code> that describes this                              
046   * <code>Student</code>.                                                          
047   */                                                                               
048  public String toString() {
049    return this.name + " has a GPA of " + formatGpa() + " and is taking " + this.classes.size() + " classes:";
050  }
051
052  private String formatGpa() {
053    NumberFormat format = new DecimalFormat("#.0#");
054    return format.format(this.gpa);
055  }
056
057  /**
058   * Main program that parses the command line, creates a
059   * <code>Student</code>, and prints a description of the student to
060   * standard out by invoking its <code>toString</code> method.
061   */
062  public static void main(String[] args) {
063    String name = null;
064    String gender = null;
065    String gpaString = null;
066
067    for (String arg : args) {
068      if (name == null) {
069        name = arg;
070
071      } else if (gender == null) {
072        gender = arg;
073
074      } else if (gpaString == null) {
075        gpaString = arg;
076      }
077    }
078
079    if (name == null) {
080      printErrorMessageAndExit("Missing command line arguments");
081
082    } else if (gender == null) {
083      printErrorMessageAndExit("Missing gender");
084
085    } else if (gpaString == null) {
086      printErrorMessageAndExit("Missing GPA");
087    }
088
089    double gpa;
090    try {
091      gpa = Double.parseDouble(gpaString);
092
093    } catch (NumberFormatException ex) {
094      printErrorMessageAndExit("Invalid GPA: " + gpaString);
095    }
096  }
097
098  private static void printErrorMessageAndExit(String message) {
099    System.err.println(message);
100    System.exit(1);
101  }
102}
