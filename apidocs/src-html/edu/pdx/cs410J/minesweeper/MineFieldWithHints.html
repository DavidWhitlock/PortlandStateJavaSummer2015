001package edu.pdx.cs410J.minesweeper;
002
003import com.google.common.annotations.VisibleForTesting;
004
005import java.util.List;
006
007public class MineFieldWithHints {
008  private char[][] grid;
009
010  public MineFieldWithHints(List<String> rows) {
011    this.grid = new char[rows.size()][rows.get(0).length()];
012
013    char[][] mineFieldMatrix = convertToCharMatrix(rows);
014
015    for (int row = 0; row < mineFieldMatrix.length; row++) {
016      for (int column = 0; column < mineFieldMatrix[0].length; column++) {
017        this.grid[row][column] = getGridCellValue(mineFieldMatrix, row, column);
018      }
019    }
020
021  }
022
023  private char getGridCellValue(char[][] mineFieldMatrix, int row, int column) {
024    if (cellContainsMine(mineFieldMatrix, row, column)) {
025      return '*';
026    }
027
028    int numberOfAdjacentMines = 0;
029
030    for (int r = row - 1; r <= row + 1; r++) {
031      for (int c = column - 1; c <= column + 1; c ++) {
032        if (cellContainsMine(mineFieldMatrix, r, c)) {
033          numberOfAdjacentMines++;
034        }
035      }
036    }
037
038    assert numberOfAdjacentMines < 9;
039    return String.valueOf(numberOfAdjacentMines).charAt(0);
040  }
041
042  private boolean cellContainsMine(char[][] mineFieldMatrix, int row, int column) {
043    if (row < 0 || column < 0) {
044      return false;
045
046    } else if (row >= mineFieldMatrix.length || column >= mineFieldMatrix[0].length) {
047      return false;
048
049    } else {
050      return mineFieldMatrix[row][column] == '*';
051    }
052  }
053
054  @VisibleForTesting
055  static char[][] convertToCharMatrix(List<String> rows) {
056    char[][] matrix = new char[rows.size()][rows.get(0).length()];
057
058    for (int rowIndex = 0; rowIndex < rows.size(); rowIndex++) {
059      String row = rows.get(rowIndex);
060      matrix[rowIndex] = row.toCharArray();
061    }
062
063    return matrix;
064  }
065
066  public char getCharAt(int row, int column) {
067    return this.grid[row][column];
068  }
069}
