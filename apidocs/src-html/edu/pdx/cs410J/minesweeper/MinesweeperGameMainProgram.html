001package edu.pdx.cs410J.minesweeper;
002
003import java.io.BufferedReader;
004import java.io.IOException;
005import java.io.InputStreamReader;
006import java.io.PrintStream;
007import java.util.ArrayList;
008import java.util.List;
009import java.util.stream.IntStream;
010
011public class MinesweeperGameMainProgram {
012
013  private final MinesweeperGame game;
014  private final BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
015
016  public MinesweeperGameMainProgram(int numberOfRows, int numberOfColumns, MinesweeperGame.MineGameGenerator generator) {
017    this.game = new MinesweeperGame(numberOfRows, numberOfColumns, generator);
018  }
019
020  public static void main(String[] args) {
021    String numberOfRowsString = null;
022    String numberOfColumnsString = null;
023    List<String> rows = new ArrayList<>();
024
025    for (String arg : args) {
026      if (numberOfRowsString == null) {
027        numberOfRowsString = arg;
028
029      } else if (numberOfColumnsString == null) {
030        numberOfColumnsString = arg;
031
032      } else {
033        rows.add(arg);
034      }
035    }
036
037    if (numberOfRowsString == null) {
038      usage("Missing number of rows");
039      return;
040    }
041
042    if (numberOfColumnsString == null) {
043      usage("Missing number of columns");
044      return;
045    }
046
047    int numberOfRows;
048    try {
049      numberOfRows = Integer.parseInt(numberOfRowsString);
050
051    } catch (NumberFormatException ex) {
052      usage("Invalid number of rows: " + numberOfRowsString);
053      return;
054    }
055
056    int numberOfColumns;
057    try {
058      numberOfColumns = Integer.parseInt(numberOfColumnsString);
059
060    } catch (NumberFormatException ex) {
061      usage("Invalid number of rows: " + numberOfColumnsString);
062      return;
063    }
064
065    MinesweeperGame.MineGameGenerator generator;
066    if (rows.isEmpty()) {
067      generator = new RandomMinesweeperGameGenerator();
068
069    } else {
070      generator = (row, column) -> rows.get(row).charAt(column);
071    }
072
073    MinesweeperGameMainProgram main = new MinesweeperGameMainProgram(numberOfRows, numberOfColumns, generator);
074    main.playGame();
075  }
076
077  private void playGame() {
078    GameState state = this.game.probe(-1, -1);
079    while(this.game.getGameResult() == GameState.GameResult.IN_PROGRESS) {
080      printGameState(state);
081
082      RowAndColumn rowAndColumn;
083      try {
084        rowAndColumn = askUserForRowAndColumn();
085
086      } catch (InvalidRowAndColumnException ex) {
087        System.out.println("Invalid input: " + ex.getLineFromInput());
088        System.out.println("  " + ex.getMessage());
089        System.out.println();
090        continue;
091      }
092
093      state = this.game.probe(rowAndColumn.getRow(), rowAndColumn.getColumn());
094    }
095
096    printGameState(state);
097
098    GameState.GameResult result = this.game.getGameResult();
099    if (result == GameState.GameResult.WON) {
100      System.out.println("Congratulations!  You won the game!");
101
102    } else if (result == GameState.GameResult.LOST) {
103      System.out.println("Bummer.  You found a mine and lost the game.");
104
105    } else {
106      System.err.println("Wait!  How did we end up in the " + result + " state?");
107    }
108
109  }
110
111  private void printGameState(GameState state) {
112    PrintStream out = System.out;
113
114    out.println();
115
116    out.print("  ");
117    IntStream.range(0, state.getNumberOfColumns()).forEach(out::print);
118    out.println();
119
120    out.print(" +");
121    IntStream.range(0, state.getNumberOfColumns()).forEach(i -> out.print("-"));
122    out.println("+");
123
124    for (int row = 0; row < state.getNumberOfRows(); row++) {
125      out.print(row + "|");
126      for (int column = 0; column < state.getNumberOfColumns(); column++) {
127        out.print(state.getCharAt(row, column));
128      }
129      out.println("|");
130    }
131
132    out.print(" +");
133    IntStream.range(0, state.getNumberOfColumns()).forEach(i -> out.print("-"));
134    out.println("+");
135  }
136
137  private RowAndColumn askUserForRowAndColumn() throws InvalidRowAndColumnException {
138    System.out.print("Enter the row and column you want to probe: ");
139    System.out.flush();
140    String line = readLineFromStandardInput();
141    return new RowAndColumn(line);
142  }
143
144  private String readLineFromStandardInput() {
145    try {
146      return in.readLine();
147
148    } catch (IOException e) {
149      e.printStackTrace();
150      System.exit(1);
151      return null;
152    }
153  }
154
155  private static void usage(String message) {
156    PrintStream err = System.err;
157    err.println("** " + message);
158    err.println();
159    err.println("Main program for playing minesweeper");
160    err.println();
161    err.println("java MinesweeperGameMainProgram numberOfRows numberOfColumns");
162    err.println();
163
164    System.exit(1);
165  }
166
167  private class RowAndColumn {
168    private final int row;
169    private final int column;
170
171    public RowAndColumn(String line) throws InvalidRowAndColumnException {
172      String[] array = line.split(" ");
173      if (array.length != 2) {
174        throw new InvalidRowAndColumnException("Missing row or column", line);
175      }
176
177      int row;
178      try {
179        row = Integer.parseInt(array[0]);
180
181      } catch (NumberFormatException ex) {
182        throw new InvalidRowAndColumnException("Row is not an integer", line);
183      }
184
185      int column;
186      try {
187        column = Integer.parseInt(array[1]);
188
189      } catch (NumberFormatException ex) {
190        throw new InvalidRowAndColumnException("Column is not an integer", line);
191      }
192
193      this.row = row;
194      this.column = column;
195
196    }
197
198    public int getRow() {
199      return row;
200    }
201
202    public int getColumn() {
203      return column;
204    }
205  }
206
207  private class InvalidRowAndColumnException extends Throwable {
208    private final String line;
209
210    public InvalidRowAndColumnException(String message, String line) {
211      super(message);
212      this.line = line;
213    }
214
215    public String getLineFromInput() {
216      return line;
217    }
218  }
219}
