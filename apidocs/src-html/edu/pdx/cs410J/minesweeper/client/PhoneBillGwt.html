001package edu.pdx.cs410J.minesweeper.client;
002
003import com.google.gwt.core.client.EntryPoint;
004import com.google.gwt.core.client.GWT;
005import com.google.gwt.user.client.Window;
006import com.google.gwt.user.client.rpc.AsyncCallback;
007import com.google.gwt.user.client.ui.DeckPanel;
008import com.google.gwt.user.client.ui.RootPanel;
009
010/**
011 * A basic GWT class that makes sure that we can send an Phone Bill back from the server
012 */
013public class PhoneBillGwt implements EntryPoint {
014
015  static final int GAME_SCREEN = 1;
016  static final int START_SCREEN = 0;
017  private DeckPanel gamePanels;
018  private MinesweeperGameServiceAsync service;
019  private GameScreen gameScreen;
020
021  @Override
022  public void onModuleLoad() {
023    this.service = GWT.create(MinesweeperGameService.class);
024
025    gamePanels = new DeckPanel();
026    gamePanels.add(createStartScreen());
027    gamePanels.add(createGameScreen());
028
029    RootPanel rootPanel = RootPanel.get();
030    rootPanel.add(gamePanels);
031    gamePanels.showWidget(START_SCREEN);
032  }
033
034  private GameScreen createGameScreen() {
035    gameScreen = new GameScreen(service);
036    return gameScreen;
037  }
038
039  private StartScreen createStartScreen() {
040    StartScreen startScreen = new StartScreen();
041    startScreen.addValidGameDimensionsListener(new StartScreen.ValidGameDimensionsListener() {
042      @Override
043      public void validGameDimensions(int rows, int columns) {
044        service.createMinesweeperGame(rows, columns, new AsyncCallback<GameState>() {
045          @Override
046          public void onFailure(Throwable throwable) {
047            Window.alert(throwable.getMessage());
048          }
049
050          @Override
051          public void onSuccess(GameState gameState) {
052            gameScreen.setGridSize(gameState.getNumberOfRows(), gameState.getNumberOfColumns());
053            gameScreen.displayGameState(gameState);
054            gamePanels.showWidget(GAME_SCREEN);
055          }
056
057        });
058      }
059    });
060    return startScreen;
061  }
062
063}
