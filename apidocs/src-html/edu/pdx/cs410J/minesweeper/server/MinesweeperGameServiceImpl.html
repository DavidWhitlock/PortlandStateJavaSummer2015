001package edu.pdx.cs410J.minesweeper.server;
002
003import com.google.gwt.user.server.rpc.RemoteServiceServlet;
004import edu.pdx.cs410J.minesweeper.client.GameState;
005import edu.pdx.cs410J.minesweeper.MinesweeperGame;
006import edu.pdx.cs410J.minesweeper.RandomMinesweeperGameGenerator;
007import edu.pdx.cs410J.minesweeper.client.MinesweeperGameService;
008
009/**
010 * The server-side implementation of the Phone Bill service
011 */
012public class MinesweeperGameServiceImpl extends RemoteServiceServlet implements MinesweeperGameService
013{
014
015  private MinesweeperGame game;
016
017  @Override
018  public GameState createMinesweeperGame(int rows, int columns) {
019    game = new MinesweeperGame(rows, columns, new RandomMinesweeperGameGenerator());
020    return game.probe(-1, -1);
021  }
022
023  @Override
024  public GameState probe(int row, int column) {
025    return game.probe(row, column);
026  }
027
028  /**
029   * Log unhandled exceptions to standard error
030   *
031   * @param unhandled
032   *        The exception that wasn't handled
033   */
034  @Override
035  protected void doUnexpectedFailure(Throwable unhandled) {
036    unhandled.printStackTrace(System.err);
037    super.doUnexpectedFailure(unhandled);
038  }
039}
