001package edu.pdx.cs410J.minesweeper;
002
003import static edu.pdx.cs410J.minesweeper.GameState.GameResult.IN_PROGRESS;
004import static edu.pdx.cs410J.minesweeper.GameState.GameResult.LOST;
005import static edu.pdx.cs410J.minesweeper.GameState.GameResult.WON;
006
007public class MinesweeperGame {
008
009  private final GameState gameState;
010  private final MineFieldWithHints mineFieldWithHints;
011
012  public MinesweeperGame(int rows, int columns, MineGameGenerator mineGameGenerator) {
013    this.gameState = new GameState(rows, columns);
014    this.mineFieldWithHints = createMineFieldWithHints(rows, columns, mineGameGenerator);
015  }
016
017  private MineFieldWithHints createMineFieldWithHints(int rows, int columns, MineGameGenerator generator) {
018    MineField field = new MineField(rows, columns);
019    for (int row = 0 ; row < rows; row++) {
020      StringBuilder line = new StringBuilder();
021      for (int column = 0; column < columns; column++) {
022        line.append(generator.getCharAt(row, column));
023      }
024      field.addRow(line.toString());
025    }
026    return field.getHints();
027  }
028
029  public GameState probe(int row, int column) {
030    if (rowAndColumnAreInBounds(row, column)) {
031      char cell = mineFieldWithHints.getCharAt(row, column);
032
033      if (cell == '0') {
034        if (this.gameState.getCharAt(row, column) != ' ') {
035          this.gameState.setCharAt(row, column, ' ');
036          probeAllAround(row, column);
037        }
038
039      } else {
040        this.gameState.setCharAt(row, column, cell);
041      }
042
043      if (cell == '*') {
044        this.gameState.setGameResult(LOST);
045        revealAllCells();
046
047      } else if (onlyMinesRemainUnprobed()) {
048        this.gameState.setGameResult(WON);
049        revealAllCells();
050
051      } else {
052        this.gameState.setGameResult(IN_PROGRESS);
053      }
054    }
055
056    return this.gameState;
057  }
058
059  private void revealAllCells() {
060    for (int row = 0; row < gameState.getNumberOfRows(); row++) {
061      for (int column = 0; column < gameState.getNumberOfColumns(); column++) {
062        char cell = mineFieldWithHints.getCharAt(row, column);
063
064        if (cell == '0') {
065          this.gameState.setCharAt(row, column, ' ');
066
067        } else {
068          this.gameState.setCharAt(row, column, cell);
069        }
070      }
071    }
072  }
073
074  private void probeAllAround(int row, int column) {
075    for (int r = row - 1; r <= row + 1; r++) {
076      for (int c = column - 1; c <= column + 1; c++) {
077        if (!(r == row && c == column)) {
078          probe(r, c);
079        }
080      }
081    }
082
083  }
084
085  private boolean rowAndColumnAreInBounds(int row, int column) {
086    return row >= 0 && column >= 0 && row < gameState.getNumberOfRows() && column < gameState.getNumberOfColumns();
087  }
088
089  private boolean onlyMinesRemainUnprobed() {
090    for (int row = 0; row < gameState.getNumberOfRows(); row++) {
091      for (int column = 0; column < gameState.getNumberOfColumns(); column++) {
092        char gameCell = gameState.getCharAt(row, column);
093        char hintCell = mineFieldWithHints.getCharAt(row, column);
094        if (gameCell == '?' && hintCell != '*') {
095          return false;
096        }
097      }
098    }
099
100    return true;
101  }
102
103  public GameState.GameResult getGameResult() {
104    return this.gameState.getGameResult();
105  }
106
107  interface MineGameGenerator {
108
109    char getCharAt(int row, int column);
110  }
111}
