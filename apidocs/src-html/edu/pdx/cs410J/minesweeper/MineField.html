001package edu.pdx.cs410J.minesweeper;
002
003import java.util.ArrayList;
004import java.util.List;
005
006/**
007 * The main class for the CS410J Phone Bill Project
008 */
009public class MineField {
010
011  private List<String> rows = new ArrayList<>();
012
013  public MineField(int rows, int columns) {
014    if (checkBounds(rows) || columns <= 0 || columns > 100) {
015      throw new IllegalArgumentException("%dx%d is not a valid dimension for a mine field");
016    }
017
018  }
019
020  private boolean checkBounds(int dimension) {
021    return dimension <= 0 || dimension > 100;
022  }
023
024  public void addRow(String line) {
025    this.rows.add(line);
026  }
027
028  public MineFieldWithHints getHints() {
029    return new MineFieldWithHints(this.rows);
030  }
031}
