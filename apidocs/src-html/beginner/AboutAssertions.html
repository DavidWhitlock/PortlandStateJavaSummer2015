001package beginner;
002
003// FYI - usually bad practice to import statically, but can make code cleaner
004import com.sandwich.koan.Koan;
005
006import static com.sandwich.util.Assert.*;
007
008public class AboutAssertions {
009
010        @Koan 
011        public void assertBooleanTrue() {
012                assertTrue(true); // should be true really
013        }
014
015        @Koan
016        public void assertBooleanFalse() {
017                assertFalse(false);
018        }
019        
020        @Koan
021        public void assertNullObject(){
022                assertNull(null);
023        }
024        
025        @Koan
026        public void assertNotNullObject(){
027                assertNotNull(new Object()); // anything other than null should pass here...
028        }
029
030        @Koan
031        public void assertEqualsUsingExpression(){
032                assertTrue("Hello World!".equals("Hello World!"));
033        }
034
035        @Koan
036        public void assertEqualsWithBetterFailureMessage(){
037                assertEquals("This should equal 1", 1, 1);
038        }
039
040        @Koan
041        public void assertEqualsWithDescriptiveMessage() {
042                // Generally, when using an assertXXX methods, expectation is on the 
043                // left and it is best practice to use a String for the first arg 
044                // indication what has failed
045                assertEquals("The answer to 'life the universe and everything' should be 42", 42, 42);
046        }
047        
048        @Koan
049        public void assertSameInstance(){
050                Object same                             = new Integer(1);
051                Object sameReference    = same;
052                assertSame(same, sameReference);
053        }
054        
055        @Koan
056        public void assertNotSameInstance(){
057                Integer same           = new Integer(1);
058                Integer sameReference  = new Integer(1);
059                assertNotSame(same, sameReference); 
060        }
061
062        @Koan
063        public void assertSameStringInstance(){
064                String same           = "Same";
065                String sameReference  = new String("Same").intern();
066                assertSame(same, sameReference);
067        }
068}
