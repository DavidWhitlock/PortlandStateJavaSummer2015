001package beginner;
002
003import com.sandwich.koan.Koan;
004
005import java.util.*;
006import java.text.MessageFormat;
007
008import static com.sandwich.koan.constant.KoanConstants.*;
009import static com.sandwich.util.Assert.*;
010
011public class AboutStrings {
012
013        @Koan
014        public void implicitStrings() {
015                assertEquals("just a plain ole string".getClass(), __);
016        }
017
018        @Koan
019        public void newString() {
020                // very rarely if ever should Strings be created via new String() in 
021                // practice - generally it is redundant, and done repetitively can be slow
022                String string = new String();
023                String empty = "";
024                assertEquals(string.equals(empty), __);
025        }
026
027        @Koan
028        public void newStringIsRedundant() {
029                String stringInstance = "zero";
030                String stringReference = new String(stringInstance);
031                assertEquals(stringInstance.equals(stringReference), __);
032        }
033
034        @Koan
035        public void newStringIsNotIdentical() {
036                String stringInstance = "zero";
037                String stringReference = new String(stringInstance);
038                assertEquals(stringInstance == stringReference, __);
039        }
040
041        @Koan
042        public void stringConcatenation() {
043                String one = "one";
044                String space = " ";
045                String two = "two";
046                assertEquals(one + space + two, __);
047        }
048
049        @Koan
050        public void stringBuilderCanActAsAMutableString() {
051                // StringBuilder concatenation looks uglier, but is useful when you need a
052                // mutable String like object. It used to be more efficient than using +
053                // to concatenate numerous strings, however this is optimized in the compiler now.
054                // Usually + concatenation is more appropriate than StringBuilder.
055                assertEquals(new StringBuilder("one").append(" ").append("two").toString(), __);
056        }
057
058        @Koan
059        public void readableStringFormattingWithStringFormat() {
060                assertEquals(String.format("%s %s %s", "a", "b", "a"), __);
061        }
062
063        @Koan
064        public void extraArgumentsToStringFormatGetIgnored() {
065                assertEquals(String.format("%s %s %s", "a", "b", "c", "d"), __);
066        }
067
068        @Koan
069        public void insufficientArgumentsToStringFormatCausesAnError() {
070                try {
071                        String.format("%s %s %s", "a", "b");
072                        fail("No Exception was thrown!");
073                } catch (Exception e) {
074                        assertEquals(e.getClass(), __);
075                        assertEquals(e.getMessage(), __);
076                }
077        }
078
079        @Koan
080        public void readableStringFormattingWithMessageFormat() {
081                assertEquals(MessageFormat.format("{0} {1} {0}", "a", "b"), __);
082        }
083
084        @Koan
085        public void extraArgumentsToMessageFormatGetIgnored() {
086                assertEquals(MessageFormat.format("{0} {1} {0}", "a", "b", "c"), __);
087        }
088
089        @Koan
090        public void insufficientArgumentsToMessageFormatDoesNotReplaceTheToken() {
091                assertEquals(MessageFormat.format("{0} {1} {0}", "a"), __);
092        }
093
094}
