001package beginner;
002
003import com.sandwich.koan.Koan;
004
005import static com.sandwich.util.Assert.assertEquals;
006
007public class AboutEquality {
008
009        @Koan
010        public void doubleEqualsTestsIfTwoObjectsAreTheSame(){
011                Object object = new Object();
012                Object sameObject = object;
013                assertEquals(object == sameObject, true);
014                assertEquals(object == new Object(), false);
015        }
016
017        @Koan
018        public void equalsMethodByDefaultTestsIfTwoObjectsAreTheSame(){
019                Object object = new Object();
020                assertEquals(object.equals(object), true);
021                assertEquals(object.equals(new Object()), false);
022        }
023
024        @Koan
025        public void equalsMethodCanBeChangedBySubclassesToTestIfTwoObjectsAreEqual(){
026                Object object = new Integer(1);
027                assertEquals(object.equals(object), true);
028                assertEquals(object.equals(new Integer(1)), true);
029                // Note: This means that for the class 'Object' there is no difference between 'equal' and 'same'
030        }
031
032        @Koan
033        public void objectsNeverEqualNull(){
034                assertEquals(new Object().equals(null), false);
035        }
036
037        @Koan
038        public void objectsEqualThemselves(){
039                Object obj = new Object();
040                assertEquals(obj.equals(obj), true);
041        }
042}
