001package beginner;
002
003
004import com.sandwich.koan.Koan;
005import static com.sandwich.koan.constant.KoanConstants.__;
006import static com.sandwich.util.Assert.assertEquals;
007
008
009public class AboutLoops {
010
011        @Koan
012        public void basicForLoop() {
013                String s = "";
014                for(int i = 0; i < 5; i++) {
015                        s += i + " ";
016                }
017                assertEquals(s, __);
018        }
019        
020        @Koan
021        public void basicForLoopWithTwoVariables() {
022                String s = "";
023                for(int i = 0, j = 10; i < 5 && j > 5; i++, j--) {
024                        s += i + " " + j + " ";
025                }
026                assertEquals(s, __);
027        }
028        
029        @Koan
030        public void extendedForLoop() {
031                int[] is = {1,2,3,4};
032                String s = "-";
033                for(int j : is) {
034                        s += "." + j;
035                }
036                assertEquals(s, __);
037        }
038        
039        @Koan
040        public void whileLoop() {
041                int result = 0;
042                while(result < 3) {
043                        result++;
044                }
045                assertEquals(result, __);
046        }
047        
048        @Koan
049        public void doLoop() {
050                int result = 0;
051                do {
052                        result++;
053                } while(false);
054                assertEquals(result, __);
055        }
056        
057        @Koan
058        public void extendedForLoopBreak() {
059                String[] sa = {"Dog", "Cat", "Tiger"};
060                int count = 0;
061                for(String current : sa) {
062                        if("Cat".equals(current)) {
063                                break;
064                        }
065                        count++;        
066                }
067                assertEquals(count, __);
068        }
069        
070        @Koan
071        public void extendedForLoopContinue() {
072                String[] sa = {"Dog", "Cat", "Tiger"};
073                int count = 0;
074                for(String current : sa) {
075                        if("Dog".equals(current)) {
076                                continue;
077                        } else {
078                                count++;        
079                        }
080                }
081                assertEquals(count, __);
082        }
083
084        @Koan
085        public void forLoopContinueLabel() {
086                int count = 0;
087                outerLabel:
088                for(int i = 0; i < 5; i++) {
089                        for(int j = 0; j < 5; j++)
090                        {
091                                count++;
092                                if(count > 2) {
093                                        continue outerLabel;    
094                                }
095                        }
096                        count += 10;
097                }
098                // What does continue with a label mean? 
099                // What gets executed? Where does the program flow continue?
100                assertEquals(count, __);
101        }
102        
103        @Koan
104        public void forLoopBreakLabel() {
105                int count = 0;
106                outerLabel:
107                for(int i = 0; i < 5; i++) {
108                        for(int j = 0; j < 5; j++)
109                        {
110                                count++;
111                                if(count > 2) {
112                                        break outerLabel;       
113                                }
114                        }
115                        count += 10;
116                }
117                // What does break with a label mean? 
118                // What gets executed? Where does the program flow continue?
119                assertEquals(count, __);
120        }
121
122}
