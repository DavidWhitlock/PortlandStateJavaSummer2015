96    /**
97     * Creates a new <code>Student</code> with the given name.
98     * @param name The name of the student
99     * @param gpa
100    * @param numClasses
101    * @return The new student
102    */
103   private Student createStudent(String name, double gpa, int numClasses) {
104     ArrayList classes = new ArrayList();
105 
106     for (int i = 0; i < numClasses; i++) {
107       classes.add("Class " + i);
108     }
109 
110     return new Student(name, classes, gpa, "male");
111   }
