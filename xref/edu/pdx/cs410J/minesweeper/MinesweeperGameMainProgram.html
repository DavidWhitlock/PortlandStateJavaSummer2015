20    public static void main(String[] args) {
21      String numberOfRowsString = null;
22      String numberOfColumnsString = null;
23      List<String> rows = new ArrayList<>();
24  
25      for (String arg : args) {
26        if (numberOfRowsString == null) {
27          numberOfRowsString = arg;
28  
29        } else if (numberOfColumnsString == null) {
30          numberOfColumnsString = arg;
31  
32        } else {
33          rows.add(arg);
34        }
35      }
36  
37      if (numberOfRowsString == null) {
38        usage("Missing number of rows");
39        return;
40      }
41  
42      if (numberOfColumnsString == null) {
43        usage("Missing number of columns");
44        return;
45      }
46  
47      int numberOfRows;
48      try {
49        numberOfRows = Integer.parseInt(numberOfRowsString);
50  
51      } catch (NumberFormatException ex) {
52        usage("Invalid number of rows: " + numberOfRowsString);
53        return;
54      }
55  
56      int numberOfColumns;
57      try {
58        numberOfColumns = Integer.parseInt(numberOfColumnsString);
59  
60      } catch (NumberFormatException ex) {
61        usage("Invalid number of rows: " + numberOfColumnsString);
62        return;
63      }
64  
65      MinesweeperGame.MineGameGenerator generator;
66      if (rows.isEmpty()) {
67        generator = new RandomMinesweeperGameGenerator();
68  
69      } else {
70        generator = (row, column) -> rows.get(row).charAt(column);
71      }
72  
73      MinesweeperGameMainProgram main = new MinesweeperGameMainProgram(numberOfRows, numberOfColumns, generator);
74      main.playGame();
75    }

77    private void playGame() {
78      GameState state = this.game.probe(-1, -1);
79      while(this.game.getGameResult() == GameState.GameResult.IN_PROGRESS) {
80        printGameState(state);
81  
82        RowAndColumn rowAndColumn;
83        try {
84          rowAndColumn = askUserForRowAndColumn();
85  
86        } catch (InvalidRowAndColumnException ex) {
87          System.out.println("Invalid input: " + ex.getLineFromInput());
88          System.out.println("  " + ex.getMessage());
89          System.out.println();
90          continue;
91        }
92  
93        state = this.game.probe(rowAndColumn.getRow(), rowAndColumn.getColumn());
94      }
95  
96      printGameState(state);
97  
98      GameState.GameResult result = this.game.getGameResult();
99      if (result == GameState.GameResult.WON) {
100       System.out.println("Congratulations!  You won the game!");
101 
102     } else if (result == GameState.GameResult.LOST) {
103       System.out.println("Bummer.  You found a mine and lost the game.");
104 
105     } else {
106       System.err.println("Wait!  How did we end up in the " + result + " state?");
107     }
108 
109   }

111   private void printGameState(GameState state) {
112     PrintStream out = System.out;
113 
114     out.println();
115 
116     out.print("  ");
117     IntStream.range(0, state.getNumberOfColumns()).forEach(out::print);
118     out.println();
119 
120     out.print(" +");
121     IntStream.range(0, state.getNumberOfColumns()).forEach(i -> out.print("-"));
122     out.println("+");
123 
124     for (int row = 0; row < state.getNumberOfRows(); row++) {
125       out.print(row + "|");
126       for (int column = 0; column < state.getNumberOfColumns(); column++) {
127         out.print(state.getCharAt(row, column));
128       }
129       out.println("|");
130     }
131 
132     out.print(" +");
133     IntStream.range(0, state.getNumberOfColumns()).forEach(i -> out.print("-"));
134     out.println("+");
135   }

167   private class RowAndColumn {
168     private final int row;
169     private final int column;
170 
171     public RowAndColumn(String line) throws InvalidRowAndColumnException {
172       String[] array = line.split(" ");
173       if (array.length != 2) {
174         throw new InvalidRowAndColumnException("Missing row or column", line);
175       }
176 
177       int row;
178       try {
179         row = Integer.parseInt(array[0]);
180 
181       } catch (NumberFormatException ex) {
182         throw new InvalidRowAndColumnException("Row is not an integer", line);
183       }
184 
185       int column;
186       try {
187         column = Integer.parseInt(array[1]);
188 
189       } catch (NumberFormatException ex) {
190         throw new InvalidRowAndColumnException("Column is not an integer", line);
191       }
192 
193       this.row = row;
194       this.column = column;
195 
196     }
197 
198     public int getRow() {
199       return row;
200     }
201 
202     public int getColumn() {
203       return column;
204     }
205   }

213     }
214 
215     public String getLineFromInput() {

217     }
