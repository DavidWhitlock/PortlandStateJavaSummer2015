8   public class MinesweeperGame {
9   
10    private final GameState gameState;
11    private final MineFieldWithHints mineFieldWithHints;
12  
13    public MinesweeperGame(int rows, int columns, MineGameGenerator mineGameGenerator) {
14      this.gameState = new GameState(rows, columns);
15      this.mineFieldWithHints = createMineFieldWithHints(rows, columns, mineGameGenerator);
16    }
17  
18    private MineFieldWithHints createMineFieldWithHints(int rows, int columns, MineGameGenerator generator) {
19      MineField field = new MineField(rows, columns);
20      for (int row = 0 ; row < rows; row++) {
21        StringBuilder line = new StringBuilder();
22        for (int column = 0; column < columns; column++) {
23          line.append(generator.getCharAt(row, column));
24        }
25        field.addRow(line.toString());
26      }
27      return field.getHints();
28    }
29  
30    public GameState probe(int row, int column) {
31      if (rowAndColumnAreInBounds(row, column)) {
32        char cell = mineFieldWithHints.getCharAt(row, column);
33  
34        if (cell == '0') {
35          if (this.gameState.getCharAt(row, column) != ' ') {
36            this.gameState.setCharAt(row, column, ' ');
37            probeAllAround(row, column);
38          }
39  
40        } else {
41          this.gameState.setCharAt(row, column, cell);
42        }
43  
44        if (cell == '*') {
45          this.gameState.setGameResult(LOST);
46          revealAllCells();
47  
48        } else if (onlyMinesRemainUnprobed()) {
49          this.gameState.setGameResult(WON);
50          revealAllCells();
51  
52        } else {
53          this.gameState.setGameResult(IN_PROGRESS);
54        }
55      }
56  
57      return this.gameState;
58    }
59  
60    private void revealAllCells() {
61      for (int row = 0; row < gameState.getNumberOfRows(); row++) {
62        for (int column = 0; column < gameState.getNumberOfColumns(); column++) {
63          char cell = mineFieldWithHints.getCharAt(row, column);
64  
65          if (cell == '0') {
66            this.gameState.setCharAt(row, column, ' ');
67  
68          } else {
69            this.gameState.setCharAt(row, column, cell);
70          }
71        }
72      }
73    }
74  
75    private void probeAllAround(int row, int column) {
76      for (int r = row - 1; r <= row + 1; r++) {
77        for (int c = column - 1; c <= column + 1; c++) {
78          if (!(r == row && c == column)) {
79            probe(r, c);
80          }
81        }
82      }
83  
84    }
85  
86    private boolean rowAndColumnAreInBounds(int row, int column) {
87      return row >= 0 && column >= 0 && row < gameState.getNumberOfRows() && column < gameState.getNumberOfColumns();
88    }
89  
90    private boolean onlyMinesRemainUnprobed() {
91      for (int row = 0; row < gameState.getNumberOfRows(); row++) {
92        for (int column = 0; column < gameState.getNumberOfColumns(); column++) {
93          char gameCell = gameState.getCharAt(row, column);
94          char hintCell = mineFieldWithHints.getCharAt(row, column);
95          if (gameCell == '?' && hintCell != '*') {
96            return false;
97          }
98        }
99      }
100 
101     return true;
102   }
103 
104   public GameState.GameResult getGameResult() {
105     return this.gameState.getGameResult();
106   }
107 
108   interface MineGameGenerator {
109 
110     char getCharAt(int row, int column);
111   }
112 }
