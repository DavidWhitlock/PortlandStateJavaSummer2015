3   import java.util.List;
4   
5   public class MineFieldWithHints {
6     private char[][] grid;
7   
8     public MineFieldWithHints(List<String> rows) {
9       this.grid = new char[rows.size()][rows.get(0).length()];
10  
11      char[][] mineFieldMatrix = convertToCharMatrix(rows);
12  
13      for (int row = 0; row < mineFieldMatrix.length; row++) {
14        for (int column = 0; column < mineFieldMatrix[0].length; column++) {
15          this.grid[row][column] = getGridCellValue(mineFieldMatrix, row, column);
16        }
17      }
18  
19    }
20  
21    private char getGridCellValue(char[][] mineFieldMatrix, int row, int column) {
22      if (cellContainsMine(mineFieldMatrix, row, column)) {
23        return '*';
24      }
25  
26      int numberOfAdjacentMines = 0;
27  
28      for (int r = row - 1; r <= row + 1; r++) {
29        for (int c = column - 1; c <= column + 1; c ++) {
30          if (cellContainsMine(mineFieldMatrix, r, c)) {
31            numberOfAdjacentMines++;
32          }
33        }
34      }
35  
36      assert numberOfAdjacentMines < 9;
37      return String.valueOf(numberOfAdjacentMines).charAt(0);
38    }
39  
40    private boolean cellContainsMine(char[][] mineFieldMatrix, int row, int column) {
41      if (row < 0 || column < 0) {
42        return false;
43  
44      } else if (row >= mineFieldMatrix.length || column >= mineFieldMatrix[0].length) {
45        return false;
46  
47      } else {
48        return mineFieldMatrix[row][column] == '*';
49      }
50    }
51  
52    static char[][] convertToCharMatrix(List<String> rows) {
53      char[][] matrix = new char[rows.size()][rows.get(0).length()];
54  
55      for (int rowIndex = 0; rowIndex < rows.size(); rowIndex++) {
56        String row = rows.get(rowIndex);
57        matrix[rowIndex] = row.toCharArray();
58      }
59  
60      return matrix;
61    }
62  
63    public char getCharAt(int row, int column) {
64      return this.grid[row][column];
65    }
66  }
