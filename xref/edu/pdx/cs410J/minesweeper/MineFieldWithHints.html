23    private char getGridCellValue(char[][] mineFieldMatrix, int row, int column) {
24      if (cellContainsMine(mineFieldMatrix, row, column)) {
25        return '*';
26      }
27  
28      int numberOfAdjacentMines = 0;
29  
30      for (int r = row - 1; r <= row + 1; r++) {
31        for (int c = column - 1; c <= column + 1; c ++) {
32          if (cellContainsMine(mineFieldMatrix, r, c)) {
33            numberOfAdjacentMines++;
34          }
35        }
36      }
37  
38      assert numberOfAdjacentMines < 9;
39      return String.valueOf(numberOfAdjacentMines).charAt(0);
40    }

42    private boolean cellContainsMine(char[][] mineFieldMatrix, int row, int column) {
43      if (row < 0 || column < 0) {
44        return false;
45  
46      } else if (row >= mineFieldMatrix.length || column >= mineFieldMatrix[0].length) {
47        return false;
48  
49      } else {
50        return mineFieldMatrix[row][column] == '*';
51      }
