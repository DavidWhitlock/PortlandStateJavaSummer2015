3   import com.google.gwt.user.server.rpc.RemoteServiceServlet;
4   import edu.pdx.cs410J.AbstractPhoneBill;
5   import edu.pdx.cs410J.whitlock.client.InvalidCustomerNameException;
6   import edu.pdx.cs410J.whitlock.client.PhoneBill;
7   import edu.pdx.cs410J.whitlock.client.PhoneCall;
8   import edu.pdx.cs410J.whitlock.client.PingService;
9   
10  /**
11   * The server-side implementation of the Phone Bill service
12   */
13  public class PingServiceImpl extends RemoteServiceServlet implements PingService
14  {
15    @Override
16    public AbstractPhoneBill ping(String customerName) throws InvalidCustomerNameException {
17      if (customerName == null || "".equals(customerName)) {
18        throw new InvalidCustomerNameException("Customer name cannot be blank");
19      }
20      PhoneBill phonebill = new PhoneBill(customerName);
21      phonebill.addPhoneCall(new PhoneCall());
22      return phonebill;
23    }
24  
25    /**
26     * Log unhandled exceptions to standard error
27     *
28     * @param unhandled
29     *        The exception that wasn't handled
30     */
31    @Override
32    protected void doUnexpectedFailure(Throwable unhandled) {
33      unhandled.printStackTrace(System.err);
34      super.doUnexpectedFailure(unhandled);
35    }
36  }
