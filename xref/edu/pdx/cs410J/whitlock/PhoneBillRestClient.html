3   import com.google.common.annotations.VisibleForTesting;
4   import edu.pdx.cs410J.web.HttpRequestHelper;

6   import java.io.IOException;
7   
8   /**
9    * A helper class for accessing the rest client.  Note that this class provides
10   * an example of how to make gets and posts to a URL.  You'll need to change it
11   * to do something other than just send key/value pairs.
12   */
13  public class PhoneBillRestClient extends HttpRequestHelper
14  {
15      private static final String WEB_APP = "phonebill";
16      private static final String SERVLET = "calls";
17  
18      private final String url;
19  
20  
21      /**
22       * Creates a client to the Phone Bil REST service running on the given host and port
23       * @param hostName The name of the host
24       * @param port The port
25       */
26      public PhoneBillRestClient( String hostName, int port )
27      {
28          this.url = String.format( "http://%s:%d/%s/%s", hostName, port, WEB_APP, SERVLET );
29      }
30  
31      /**
32       * Returns all keys and values from the server
33       */
34      public Response getAllKeysAndValues() throws IOException
35      {
36          return get(this.url );
37      }
38  
39      /**
40       * Returns all values for the given key
41       */
42      public Response getValues( String key ) throws IOException
43      {
44          return get(this.url, "key", key);
45      }
46  
47      public Response addKeyValuePair( String key, String value ) throws IOException
48      {
49          return postToMyURL("key", key, "value", value);
50      }
51  
52      @VisibleForTesting
53      Response postToMyURL(String... keysAndValues) throws IOException {
54          return post(this.url, keysAndValues);
55      }
56  }
