1   package edu.pdx.cs410J.whitlock.client;
2   
3   import com.google.gwt.core.client.EntryPoint;
4   import com.google.gwt.core.client.GWT;
5   import com.google.gwt.event.dom.client.ClickEvent;
6   import com.google.gwt.event.dom.client.ClickHandler;
7   import com.google.gwt.user.client.Window;
8   import com.google.gwt.user.client.rpc.AsyncCallback;
9   import com.google.gwt.user.client.ui.Button;
10  import com.google.gwt.user.client.ui.RootPanel;
11  import com.google.gwt.user.client.ui.TextBox;
12  import edu.pdx.cs410J.AbstractPhoneBill;
13  import edu.pdx.cs410J.AbstractPhoneCall;
14  
15  import java.util.Collection;
16  
17  /**
18   * A basic GWT class that makes sure that we can send an Phone Bill back from the server
19   */
20  public class PhoneBillGwt implements EntryPoint {
21  
22    private TextBox customerNameField;
23  
24    public void onModuleLoad() {
25      customerNameField = new TextBox();
26      customerNameField.setMaxLength(10);
27  
28  
29      Button button = new Button("Ping Server");
30      button.addClickHandler(createNewPhoneBillOnServer());
31  
32      RootPanel rootPanel = RootPanel.get();
33      rootPanel.add(button);
34      rootPanel.add(customerNameField);
35    }
36  
37    private ClickHandler createNewPhoneBillOnServer() {
38      return new ClickHandler() {
39        @Override
40        public void onClick(ClickEvent clickEvent) {
41          String customerName = customerNameField.getText();
42  
43          PingServiceAsync async = GWT.create(PingService.class);
44          async.ping(customerName, displayPhoneBill());
45        }
46  
47        private AsyncCallback<AbstractPhoneBill> displayPhoneBill() {
48          return new AsyncCallback<AbstractPhoneBill>() {
49  
50            @Override
51            public void onFailure(Throwable ex) {
52              if (ex instanceof InvalidCustomerNameException) {
53                Window.alert("Please enter a valid customer name");
54  
55              } else {
56                Window.alert("onFailure: " + ex.toString());
57              }
58            }
59  
60            @Override
61            public void onSuccess(AbstractPhoneBill phonebill) {
62              StringBuilder sb = new StringBuilder(phonebill.toString());
63              Collection<AbstractPhoneCall> calls = phonebill.getPhoneCalls();
64              for (AbstractPhoneCall call : calls) {
65                sb.append(call);
66                sb.append("\n");
67              }
68              Window.alert(sb.toString());
69            }
70          };
71        }
72      };
73    }
74  }
